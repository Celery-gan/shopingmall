安装依赖

```javascript

cnpm i -S axios  

cnpm i -S dayjs

cnpm i -S qs 
// 曲线图
cnpm i v-charts echarts -S 
//导出excel
cnpm install vue-json-excel -S 
//element组件
cnpm i element-ui -S
// markdown 编辑组件
cnpm install mavon-editor --save 
// lodash 库
cnpm i -S lodash 

// better-scroll(移动端页面平滑滚动)
cnpm i better-scroll -S
// fastclick(取消移动端点击延迟)
cnpm i -S fastclick
// vant(ui组件库)
cnpm i vant -S
```
